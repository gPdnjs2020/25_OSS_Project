This example demonstrates reading and writing students' scores to a file using C's standard I/O functions.

#include <stdio.h>

struct Student {
    char name[20];
    int score;
};

int main() {
    struct Student s = {"Alice", 90};
    FILE *fp = fopen("students.txt", "w");
    fprintf(fp, "%s %d\n", s.name, s.score);
    fclose(fp);
    printf("Saved & read from file complete.\n");
    return 0;
}

Run: gcc student.c -o student && ./student

This example demonstrates using a vector of Student objects and class methods in C++.

#include <iostream>
#include <vector>
using namespace std;

class Student {
    string name;
    int score;
public:
    Student(string n, int s) : name(n), score(s) {}
    void show() { cout << name << " : " << score << endl; }
};

int main() {
    vector<Student> students = {Student("Alice",90), Student("Bob",85)};
    for (auto &s : students) s.show();
    return 0;
}

Run: g++ student.cpp -o student && ./student

This example shows how to use a HashMap to manage a simple todo list in Java.

import java.util.*;

public class TodoApp {
    public static void main(String[] args) {
        HashMap<Integer,String> todos = new HashMap<>();
        todos.put(1,"Finish homework");
        todos.put(2,"Review C++ code");
        for(int key : todos.keySet()) {
            System.out.println(key + ". " + todos.get(key));
        }
    }
}

Run: javac TodoApp.java && java TodoApp
